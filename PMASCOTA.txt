CREATE TABLE USUARIO
(
	RUT INT NOT NULL,
	DV NVARCHAR(1) NOT NULL,
	NOMBRE NVARCHAR(30) NOT NULL,
	TELEFONO INT ,
	CORREO NVARCHAR(30) NOT NULL,
	USE_NAME NVARCHAR(12) NOT NULL,
	PASSWORD NVARCHAR(10) NOT NULL,
	REGION NVARCHAR(60) NOT NULL,
	COMUNA NVARCHAR(60) NOT NULL,
	DIRECCION NVARCHAR(200) NOT NULL,
	TIENE_MACOTA BIT NOT NULL,
	FECHA_NACIMIENTO DATETIME NOT NULL,

	PRIMARY KEY (RUT)
)


CREATE TABLE MASCOTA
(
	ID_MAS INT NOT NULL,
	RAZA NVARCHAR(30) NOT NULL,
	EDAD INT ,
	ESPECIE NVARCHAR(30) NOT NULL,
	NOMBRE NVARCHAR(12) NOT NULL,
	TAMAÑO FLOAT NOT NULL

	PRIMARY KEY (ID_MAS)
)

CREATE TABLE PUBLICACION
(
	ID_PUB INT NOT NULL,
	FECHA DATE NOT NULL,
	FOTO IMAGE 

	PRIMARY KEY (ID_PUB)
)

CREATE TABLE ADOPCION
(
	ID_ADOP INT NOT NULL,
	FECHA_ADOP DATE NOT NULL,

	PRIMARY KEY (ID_ADOP)
)

CREATE TABLE INVENTARIO_REGISTROS
(
	ID_REG INT NOT NULL,
	CAN_ESPECIE INT NOT NULL,
	TIPO_REG  NVARCHAR(12) NOT NULL,

	PRIMARY KEY (ID_REG)
)

CREATE TABLE COMUNA(
	IDCOMUNA int NOT NULL,
	NOMBRECOMUNA nvarchar(40) NOT NULL)

CREATE TABLE REGION(
	IDREGION int NOT NULL,
	NOMBRE_REGION nvarchar(40) NOT NULL)
